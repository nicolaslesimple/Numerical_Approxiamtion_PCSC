72 void FFTreal::transformRadix2(vector<double> &real, vector<double> &imag) {
73  // Length variables
74  size_t n = real.size();
75  if (n != imag.size())
76  throw "Mismatched lengths";
77  int levels = 0; // Compute levels = floor(log2(n))
78  for (size_t temp = n; temp > 1U; temp >>= 1)
79  levels++;
80  if (static_cast<size_t>(1U) << levels != n)
81  throw "Length is not a power of 2";
82 
83  // Trignometric tables
84  vector<double> cosTable(n / 2);
85  vector<double> sinTable(n / 2);
86  for (size_t i = 0; i < n / 2; i++)
87  {
88  cosTable[i] = std::cos(2 * M_PI * i / n);
89  sinTable[i] = std::sin(2 * M_PI * i / n);
90  }
91 
92  // Bit-reversed addressing permutation
93  for (size_t i = 0; i < n; i++) {
94  size_t j = reverseBits(i, levels);
95  if (j > i) {
96  std::swap(real[i], real[j]);
97  std::swap(imag[i], imag[j]);
98  }
99  }
100 
101  // Cooley-Tukey decimation-in-time radix-2 FFT
102  for (size_t size = 2; size <= n; size *= 2) {
103  size_t halfsize = size / 2;
104  size_t tablestep = n / size;
105  for (size_t i = 0; i < n; i += size) {
106  for (size_t j = i, k = 0; j < i + halfsize; j++, k += tablestep) {
107  size_t l = j + halfsize;
108  double tpre = real[l] * cosTable[k] + imag[l] * sinTable[k];
109  double tpim = -real[l] * sinTable[k] + imag[l] * cosTable[k];
110  real[l] = real[j] - tpre;
111  imag[l] = imag[j] - tpim;
112  real[j] += tpre;
113  imag[j] += tpim;
114  }
115  }
116  if (size == n) // Prevent overflow in 'size *= 2'
117  break;
118  }
119 }

121 void FFTreal::transformBluestein(vector<double> &real, vector<double> &imag) {
122  // Find a power-of-2 convolution length m such that m >= n * 2 + 1
123  size_t n = real.size();
124  if (n != imag.size())
125  throw "Mismatched lengths";
126  size_t m = 1;
127  while (m / 2 <= n) {
128  if (m > SIZE_MAX / 2)
129  throw "Vector too large";
130  m *= 2;
131  }
132 
133  // Trigonometric tables
134  vector<double> cosTable(n), sinTable(n);
135  for (size_t i = 0; i < n; i++) {
136  unsigned long long temp = static_cast<unsigned long long>(i) * i;
137  temp %= static_cast<unsigned long long>(n) * 2;
138  double angle = M_PI * temp / n;
139  // Less accurate alternative if long long is unavailable: double angle = M_PI * i * i / n;
140  cosTable[i] = std::cos(angle);
141  sinTable[i] = std::sin(angle);
142  }
143 
144  // Temporary vectors and preprocessing
145  vector<double> areal(m), aimag(m);
146  for (size_t i = 0; i < n; i++) {
147  areal[i] = real[i] * cosTable[i] + imag[i] * sinTable[i]; //an
148  aimag[i] = -real[i] * sinTable[i] + imag[i] * cosTable[i];
149  }
150  vector<double> breal(m), bimag(m);
151  breal[0] = cosTable[0];
152  bimag[0] = sinTable[0];
153  for (size_t i = 1; i < n; i++) {
154  breal[i] = breal[m - i] = cosTable[i]; //bn
155  bimag[i] = bimag[m - i] = sinTable[i];
156  }
157 
158  // Convolution between a and b is put in c
159  vector<double> creal(m), cimag(m);
160  convolve(areal, aimag, breal, bimag, creal, cimag);
161 
162  // Postprocessing
163  for (size_t i = 0; i < n; i++) {
164  real[i] = creal[i] * cosTable[i] + cimag[i] * sinTable[i];
165  imag[i] = -creal[i] * sinTable[i] + cimag[i] * cosTable[i];
166  }
